Я использую библиотеку better-auth в своем приложении на Next.js 16. Вот ссылка для тебя: "https://better-auth.com/llms.txt". И ещё одна ссылка для тебя: "https://mcp.better-auth.com/mcp". У меня есть файл auth.ts: "import { betterAuth } from "better-auth";
import { prismaAdapter } from "better-auth/adapters/prisma";
import prisma from "@/lib/prisma";

export const auth = betterAuth({
  database: prismaAdapter(prisma, {
    provider: "postgresql",
  }),

  baseURL: process.env.BETTER_AUTH_URL,

  socialProviders: {
    google: {
      clientId: process.env.GOOGLE_CLIENT_ID!,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET!,

      // всегда показывать выбор аккаунта
      prompt: "select_account",
    },
  },
});
". Помоги добавить пользователя - администратора. И помоги с доработкой страницы администратора: "import { auth } from "@/lib/auth";
import { headers } from "next/headers";
import { redirect } from "@/i18n/navigation";

export default async function DashboardPage() {
  const session = await auth.api.getSession({
    headers: await headers(),
  });

  if (!session) {
    redirect({
      href: "/sign-up",
      locale: "en",
    });
  }

  return <div>Dashboard</div>;
}
